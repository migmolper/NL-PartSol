123 double GetMinElementSize(Mesh FEM_Mesh)
124 /*
125  Function to get the minimum mesh size.
126 */
127 {

135  X_eval.nV[0] = 0;
136  X_eval.nV[1] = 0;

138  double MinElementSize_aux;
139  double MinElementSize = 10e16;
140 
141  /* 1ยบ Loop over the elements in the mesh */
142  for(int i = 0 ; i<NumElemMesh ; i++){
143

145  NumNodesElem = FEM_Mesh.NumNodesElem[i];
146  Connectivity = Set_to_Pointer(FEM_Mesh.Connectivity[i],NumNodesElem);
147 
148  /* 4ยบ Get the gradient of the element for each node */
149  if((NumNodesElem == 3) &&
150  (NumberDimensions == 2)){ /* Triangular element */
151  /* The poligon is a triangle */
152  Poligon = MatAllocZ(3,2);
153  /* Fill the triangle */
154  for(int k = 0; k<3; k++){
155  for(int l = 0 ; l<2 ; l++){
156  Poligon.nM[k][l] = FEM_Mesh.Coordinates.nM[Connectivity[k]][l];
157  }
158  }

160  dNdx = Get_dNdX_T3(X_eval,Poligon);
161  FreeMat(Poligon);
162

164  for(int j = 0 ; j<3 ; j++){
165  MinElementSize_aux =
166  1/pow(dNdx.nM[0][j]*dNdx.nM[0][j] +
167  dNdx.nM[1][j]*dNdx.nM[1][j],0.5);
168  MinElementSize = MINVAL(MinElementSize,MinElementSize_aux);
169  }
170  /* Free memory */
171  FreeMat(dNdx);
172 
173  }
174  else if((NumNodesElem == 4) &&
175  (NumberDimensions == 2)){ /* Quadrilateral element */
176  /* The poligon is a quadrilateral */
177  Poligon = MatAllocZ(4,2);
178 
179  /* Fill the poligon with vectors */
180  for(int k = 0; k<3; k++){
181  for(int l = 0 ; l<2 ; l++){
182  Poligon.nM[k][l] =
183  FEM_Mesh.Coordinates.nM[Connectivity[k+1]][l] -

410 ChainPtr DiscardElements(ChainPtr SearchElem_0,
411  Matrix Corner_MAX,
412  Matrix Corner_MIN,
413  Mesh FEM_Mesh)
414 /*
415  Auxiliary function to increase the computational accuracy in the
416  GlobalSearchGaussPoints() function. We discard the elements where we can't have
417  GPs because its vertex coordinates.
418 */
419 {
420  /* Found the tail */
421  if (SearchElem_0 == NULL) {
422  return NULL;
423  }
424 
425  ChainPtr SearchElem_1;
426  double Xmax_E, Ymax_E;
427  double Xmin_E, Ymin_E;
428  int Idx_Elem; /* Index of the element */
429  int NumNodes; /* Num nodes of the element */
430  int * Conect_Elem; /* Conectivity of the element */
431 
432  Idx_Elem = SearchElem_0->I;
433  NumNodes = FEM_Mesh.NumNodesElem[Idx_Elem];
434  Conect_Elem = Set_to_Pointer(FEM_Mesh.Connectivity[Idx_Elem],NumNodes);
435 
436  /* Init Corner_MAX and Corner_MIN */
437  Xmax_E = FEM_Mesh.Coordinates.nM[Conect_Elem[0]][0];
438  Ymax_E = FEM_Mesh.Coordinates.nM[Conect_Elem[0]][1];
439  Xmin_E = FEM_Mesh.Coordinates.nM[Conect_Elem[0]][0];
440  Ymin_E = FEM_Mesh.Coordinates.nM[Conect_Elem[0]][1];
441 
442  for(int i = 1 ; i<NumNodes ; i++){
443  /* Corner_MAX */
444  Xmax_E = MAXVAL(Xmax_E, FEM_Mesh.Coordinates.nM[Conect_Elem[i]][0]);
445  Ymax_E = MAXVAL(Ymax_E, FEM_Mesh.Coordinates.nM[Conect_Elem[i]][1]);
446  /* Corner_MIN */
447  Xmin_E = MINVAL(Xmin_E, FEM_Mesh.Coordinates.nM[Conect_Elem[i]][0]);
448  Ymin_E = MINVAL(Ymin_E, FEM_Mesh.Coordinates.nM[Conect_Elem[i]][1]);
449  }
450 
451  /* Free */
452  free(Conect_Elem);
453 
454  /* Test inferior limit */
455  if ((Xmin_E<Corner_MIN.nV[0]) &&
456  (Ymin_E<Corner_MIN.nV[1])){
457  SearchElem_1 = SearchElem_0->next;
458  free(SearchElem_0);
459  SearchElem_1->next =
460  DiscardElements(SearchElem_1->next,

472  Corner_MAX,Corner_MIN,
473  FEM_Mesh);
474  return SearchElem_1;
475  }
476  /* Just keep going */
477  else{
478  SearchElem_0->next =
479  DiscardElements(SearchElem_0->next,
480  Corner_MAX,Corner_MIN,
481  FEM_Mesh);
482  return SearchElem_0;
483  }
